197    def issue_command(self, command: str) -> str:
198        """
199        Issue a single command to the sensor and return the response.
200        This will add or strip end characters as required from the returning string.
201
202        :param command: Single Coimmand from 'USART Command Set'
203        :type command: str
204        :raises CommsError: If Error is returned exception will be raised/
205        :return: _description_
206        :rtype: str
207        """
208        self.write(f"{command}\r\n".encode())
209        response = self.readline().decode().strip()
210        if response[0] == "E" or not response:
211            raise CommsError(response)
212        else:
213            return response
214
215    def get_stream_raw(self) -> str:
216        """
217        Returns the raw stream from the sensor
218
219        :return: Single stream string
220        :rtype: str
221        """
222        if self.output_mode != OutputMode.STREAM:
223            self.output_mode = OutputMode.STREAM
224
225        self.reset_input_buffer()
226        return self.readline().decode()
227
228    def get_stream_decoded(self) -> dict[str : float | str]:
229        """
230        Decodes the streamed output from the sensor into a dictionary
231
232        :return: Decoded stream
233        :rtype: dict[str: float|str]
234        """
235        if self.output_mode != OutputMode.STREAM:
236            self.output_mode = OutputMode.STREAM
237
238        self.reset_input_buffer()
239        line = self.readline().decode().strip()
240        split_line = line.split(" ")
241        line_dictionary = {}
242        for index, i in enumerate(split_line):
243            if index % 2 != 0:
244                continue
245            if index + 1 == len(split_line):
246                break
247            if i == "e":
248                line_dictionary[i] = split_line[index + 1]
249            else:
250                line_dictionary[i] = float(split_line[index + 1])
251        return line_dictionary
